```ts
import { render, fireEvent } from "@testing-library/react";
import { Provider, useDispatch } from "react-redux";
import { Dispatch } from "redux";
import MyComponent from "./MyComponent";
import { store } from "./store"; // assuming you export your store

// Mocking react-redux module
jest.mock("react-redux", () => {
  const originalModule = jest.requireActual("react-redux");
  return {
    ...originalModule,
    useDispatch: jest.fn(),
  };
});

test("it dispatches the expected action on button click", () => {
  const mockDispatch = jest.fn();
  (useDispatch as jest.Mock<Dispatch<any>>).mockReturnValue(mockDispatch); // Properly typed mock function

  const { getByText } = render(
    <Provider store={store}>
      <MyComponent />
    </Provider>
  );
  fireEvent.click(getByText("Click Me"));

  // Now assert that the dispatch was called with the expected action
  expect(mockDispatch).toHaveBeenCalled(); // Check if dispatch was called
});

test("clear mockDispatch for each test", () => {
  mockDispatch.mockClear();
});
```
